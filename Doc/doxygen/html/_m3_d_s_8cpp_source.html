211 static int mcomp(const void* a, const void* b)
212 {
213  Poly* pa = (Poly*) a;
214  Poly* pb = (Poly*) b;
215 
216  if (pa->sortval == pb->sortval)
217  return 0;
218 
219  if (pa->sortval < pb->sortval)
220  return 1;
221 
222  return -1;
223 }
