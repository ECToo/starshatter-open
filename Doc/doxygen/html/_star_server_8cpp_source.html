380

382  Sleep(30);
383  }
384 
385  else if (game_mode == LOAD_MODE) {
386  CreateWorld();
387  InstantiateMission();

389  SetGameMode(PLAY_MODE);

391 
392  else if (game_mode == PLAY_MODE) {
393  if (Game::GameTime() - time_mark > 60000) {
394  time_mark = Game::GameTime();
395  minutes++;

401

403  }
404
